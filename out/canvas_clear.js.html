/**
 * Simplification for CanvasRenderingContext2D clearRect.
 * Create a clearing rectangle from the top-left to the bottom-right of the canvas element, clearing the whole canvas.
 *
 * @memberof HTMLCanvasElement
 *
 * @example <caption>Clear the canvas:</caption>
 * HTMLCanvasElement.clear(); // clear the whole canvas:
 *
 */
HTMLCanvasElement.prototype.clear = function() { 
  const context = this.getContext('2d');              // Get this canvas' element CanvasRenderingContext2D object.
  context.clearRect( 0, 0, this.width, this.height);  // Create the clearing rectangle.
}
