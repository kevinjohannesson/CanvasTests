/**
 * Draw a circle on the canvas.
 * Simplification for arc():.
 *
 * @memberof HTMLCanvasElement
 *
 * @param {Object} vectorOrigin - A { x, y } location vector.
 * @param {number} radius - The radius of the circle.
 * @param {Object} config - Changes to the default stroke and fill style.
 *
 * @todo create examples
 * @todo finish this with comments
 */
HTMLCanvasElement.prototype.drawCircle = function( vectorOrigin = { x: 300, y: 300 }, radius = 100, config = this.getContext('2d').defaultConfig, circleStart = 0, circleEnd = Math.PI * 2 ) {
  this.ctx.beginPath(); // Start the path:
  this.ctx.arc( vectorOrigin.x, vectorOrigin.y, radius, circleStart, circleEnd ); // Create the arc.
  this.draw( config ); // Stroke and or fill the path.
}
